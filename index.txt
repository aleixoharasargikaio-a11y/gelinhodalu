<script>
    const itens = [
        { n: "Nutella Morango", d: "Base de morango com 10g de Nutella." },
        { n: "Ninhotella", d: "Base de Ninho recheada com 10g de Nutella." },
        { n: "Brigadeiro", d: "Base de Ninho, chocolate e 10g de granulado macio." },
        { n: "Ninho Oreo", d: "Base de Ninho com 10g de biscoito Oreo." },
        { n: "Ninho Kat", d: "Base de Ninho com chocolate Kit Kat." },
        { n: "Briga Oreo", d: "Base chocolate, granulado macio e biscoito Oreo." },
        { n: "Nutella Morango Ninho", d: "Base morango, recheio Nutella e 10g de Ninho em pó." },
        { n: "Açaí Leite Condensado", d: "Base com Açaí e Leite Condensado." },
        { n: "Açaí Ninho Leite Condensado", d: "Base Açaí, Leite Condensado e Ninho em pó." }
    ];

    let cart = {};

    function loadMenu() {
        document.getElementById('menu-list').innerHTML = itens.map(p => `
            <div class="product-row">
                <div style="flex:1">
                    <div style="font-weight:800; font-size:0.95rem;">${p.n}</div>
                    <div style="font-size:0.75rem; color:#8d6e63;">${p.d}</div>
                    <div style="color:var(--primary); font-weight:800; font-size:0.85rem; margin-top:4px;">R$ 5,00</div>
                </div>
                <div class="qty-controls">
                    <button class="btn-action" onclick="updateQty('${p.n}', -1)">-</button>
                    <span id="q-${p.n}" style="font-weight:800; min-width:20px; text-align:center;">0</span>
                    <button class="btn-action plus" onclick="updateQty('${p.n}', 1)">+</button>
                </div>
            </div>
        `).join('');
    }

    function updateQty(name, val) {
        cart[name] = Math.max(0, (cart[name] || 0) + val);
        document.getElementById(`q-${name}`).innerText = cart[name];
        let total = Object.values(cart).reduce((a,b) => a+b, 0) * 5;
        document.getElementById('display-total').innerText = `R$ ${total.toFixed(2).replace('.', ',')}`;
    }

    function spawnSpark(e) {
        const spark = document.createElement('div');
        spark.className = 'click-spark';
        spark.style.left = `${e.pageX - 7}px`;
        spark.style.top = `${e.pageY - 7}px`;
        document.body.appendChild(spark);
        setTimeout(() => spark.remove(), 600);
    }

    function enviarWhatsApp() {
        const nome = document.getElementById('nome').value;
        const end = document.getElementById('endereco').value;
        const total = document.getElementById('display-total').innerText;
        let pedido = "";
        for(let s in cart) if(cart[s] > 0) pedido += `• ${s} (${cart[s]}x)\n`;
        if(!nome || !end || !pedido) return alert("Preencha todos os campos!");
        let texto = `*PEDIDO GELINHO DA LU*\n\n${pedido}\n*CLIENTE:* ${nome}\n*END:* ${end}\n*TOTAL:* ${total}`;
        window.open(`https://wa.me/5518981362237?text=${encodeURIComponent(texto)}`);
    }

    loadMenu();
</script>
